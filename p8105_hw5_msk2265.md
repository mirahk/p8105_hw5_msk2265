``` r
set.seed(1)

#creating function
power = function(mu, n=30, sigma = 5){
  sim_data = tibble(
    x = rnorm(n, mean = mu, sd = sigma),
  )
  
  sim_data %>%  
    summarize(
      mu,
      mu_hat = mean(x),
      ttest = broom::tidy(t.test(x, mu=mu, conf.level=.95))
    )
}
```

``` r
#creating simulations
true_mean_df = c(0,1,2,3,4,5,6)

for (i in 1:5000) {

}
```
